# 主要继承方式

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <meta http-equiv="X-UA-Compatible" content="ie=edge" />
    <title>js主要继承方式</title>
  </head>

  <body>
    <!-- 
        种类：
        1.替换原型继承
        2.函数继承（call/apply）
        3.混合继承
     -->
    <div id="app"></div>
    <script>
      //第1种.替换原型继承

      //在构造函数实例化对象前
      //1.将构造函数.prototype = 新原型
      //2.将新原型.constructor = 构造函数
      var Create = function() {
        this.name = 'aaa',
        this.sex = '男',
        this.say = function() {
            console.log('come on');
        };
      };
      //w对象拥有默认原型
      var w = new Create();
      //替换原型
      var obj = {
        age: 35
      };
      Create.prototype = obj;
      obj.constructor = Create;

      var ww = new Create();
      console.log(w.age);
      console.log(ww.age);

      //第2种.函数继承
      var Create = function() {
        this.name = 'ww';
        this.job = '东海第二技师';
      };
      var Age = function() {
        this.age = 35;
      };
      var wbb = new Create();
      Age.call(wbb);
      //通过call或者apply将函数的this指向修改为对象
      //函数中给this绑定的属性 就会被绑定到对象上
      console.log(wbb);

      //第3种.混合继承 构造函数和原型共同给对象继承
      //原则：易变化的属性由构造函数继承 不变的属性由原型继承
      var CreateM = function(name, age) {
        this.name = name;
        this.age = age;
      };
      CreateM.prototype.belong = '数字媒体学院';
      var wls = new CreateM('大师', 35);
      console.log(wls.belong);
    </script>
  </body>
</html>
