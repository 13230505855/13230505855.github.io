    # node 增删改查

    # 一、增加数据方法：

    const fs = require('fs').promises;

const path = require('path');
let rdir = filename => path.resolve(__dirname, filename);

//封装读取json的方法
function rfile(file) {
  //读取json文件 并处理json成为数组 返回一个promise
  return fs.readFile(rdir(file), 'utf8').then(data => JSON.parse(data));
}
// rfile("goods.json").then(data => {
//     console.log(data, data instanceof Array)
// })

//封装写入json的方法
function wfile(file, data) {
  if (!data) return;
  return fs.writeFile(rdir(file), JSON.stringify(data), 'utf8');
}

//添加
rfile('goods.json').then(data => {
  let newArr = [
    {
      name: '草莓',
      price: 24,
    },
    {
      name: '西瓜',
      price: 18,
    },
  ];

  let ns = newArr.filter(item => {
    let s = data.find(it => it.name == item.name);
    console.log(s, '------');
    if (s) {
      return false;
    } else {
      return item;
    }
  });
  ns = ns.map((item, index) => {
    item.id = data[data.length - 1].id + index + 1;
    return item;
  });
  //合并数组
  let lastData = [...data, ...ns];
  wfile('goods.json', lastData).then(() => {
    console.log('写入成功');
  });
});

    # 三、修改数据方法：

    const fs = require('fs').promises;

const path = require('path');
let resolve = filename => path.resolve(__dirname, filename);

//1.读取文件中的数据
fs.readFile(resolve('goods.json'), 'utf8').then(data => {
  //2.把json数据转化成数组形式
  let goodsArr = JSON.parse(data);

  //3.操作数组
  let newArr = goodsArr.map(item => {
    if (item.id == 1) {
      item.price = 20;
    }
    return item;
  });

  //4.把新数组写入文件
  fs.writeFile(resolve('goods.json'), JSON.stringify(newArr), 'utf8');
});
