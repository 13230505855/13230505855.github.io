# node 基础

# 获取服务端文件路径：

const path = require('path');

# 举例：显示绝对路径

d:\github\nodelesson

# 获取绝对路径的两个方法：

path.resolve();
path.join();

# 封装获取文件绝对路径的函数：

let resolve = filename => path.resolve(__dirname, filename);
console.log(resolve('app.js', '函数路径'));

# 注意事项：

在node里面，this是global，直接定义的变量不会在全局对象上显示

# 创建服务器：

http
  .createServer(function(request, response) {
     response.end('hello world');
  })
  .listen(8080, function() {
