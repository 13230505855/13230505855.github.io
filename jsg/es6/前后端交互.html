    # 前后端交互

    # ajax 请求方法：

    # 1.$.ajax()返回其创建的 XMLHttpRequest 对象。

    $.ajax() 只有一个参数：参数 key/value 对象，包含各配置及回调函数信息。
$.ajax({
    type: "post",
    dataType: "html",
    url: '请求地址',
    data: 参数,
    success: function (data) {   //data回调信息
        if (data != "") {
            console.log(data);
        }else{}
    }
});

    # 2.通过远程 Http Get 请求载入信息

    # 这是一个简单的 GET 请求功能以取代复杂 $.ajax 。

    # 请求成功时可调用回调函数。如果需要在出错时执行函数，请使用 $.ajax。

    $.get(
  '请求地址',
  { name: 'John', time: '2pm' }, //参数列表
  function(data) {
    alert(data);
  },
);
